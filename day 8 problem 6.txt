#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct product
{
    char pname[20];
    float price;
};

struct order
{
    int orderId;
    struct product p;
    char status[15];
    struct order *next;
};

int main()
{
    struct order *head, *second;

    head = (struct order *)malloc(sizeof(struct order));
    second = (struct order *)malloc(sizeof(struct order));

    head->orderId = 201;
    strcpy(head->p.pname, "Mouse");
    head->p.price = 500;
    strcpy(head->status, "Delivered");
    head->next = second;

    second->orderId = 202;
    strcpy(second->p.pname, "Keyboard");
    second->p.price = 1500;
    strcpy(second->status, "Pending");
    second->next = NULL;

    struct order *temp = head;

    while (temp != NULL)
    {
        printf("%d %s %.0f %s -> ", temp->orderId, temp->p.pname, temp->p.price, temp->status);
        temp = temp->next;
    }

    printf("NULL");

    return 0;
}
