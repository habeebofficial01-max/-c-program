#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Contact {
    char phone[15];
};

struct Customer {
    int accNo;
    struct Contact contact;
    struct Customer* next;
};

int main() {
    struct Customer* head = NULL;
    struct Customer* second = NULL;

    head = (struct Customer*)malloc(sizeof(struct Customer));
    second = (struct Customer*)malloc(sizeof(struct Customer));

    head->accNo = 101;
    strcpy(head->contact.phone, "9876543210");
    head->next = second;

    second->accNo = 102;
    strcpy(second->contact.phone, "9123456789");
    second->next = NULL;

    struct Customer* ptr = head;
    printf("Customer List:\n");
    while (ptr != NULL) {
        printf("%d - %s", ptr->accNo, ptr->contact.phone);
        if (ptr->next != NULL)
            printf(" -> ");
        ptr = ptr->next;
    }
    printf(" -> NULL\n");

    free(head);
    free(second);

    return 0;
}
