#include <stdio.h>
#define size 5 
 int array[size];
 int top=-1;
 //push
 void push(int value){
 if(top==size-1){
     printf("no space left \n");}
 else{
 top++;
 array[top]=value;
 printf("the value %d is pushed array \n",array[top]);}
 }
 //pop
 void pop(){
     if(top==-1){
         printf("no value at stack");
     }
     else{
         printf("the top value is: %d \n",array[top]);
         top--;
     }
 }
 void peak(){
     if(top==-1){
         printf("no value at stack");
     }
     else{
         printf("the peak value is: %d \n",array[top]);
