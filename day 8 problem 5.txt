#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct details
{
    char name[20];
    char dept[10];
};

struct node
{
    int roll;
    struct details d;
    struct node *next;
};

int main()
{
    struct node *head, *second;

    head = (struct node *)malloc(sizeof(struct node));
    second = (struct node *)malloc(sizeof(struct node));

    head->roll = 1;
    strcpy(head->d.name, "Arun");
    strcpy(head->d.dept, "CSE");
    head->next = second;

    second->roll = 2;
    strcpy(second->d.name, "Meena");
    strcpy(second->d.dept, "IT");
    second->next = NULL;

    struct node *temp = head;

    while (temp != NULL)
    {
        printf("%d %s %s -> ", temp->roll, temp->d.name, temp->d.dept);
        temp = temp->next;
    }

    printf("NULL");

    return 0;
}
